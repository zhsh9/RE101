db_utils.php

function check_auth($player, $password) {
    global $pdo;
    $params = ["player" => $player];
    $stmt = $pdo->prepare("SELECT password FROM players WHERE username = :player");
    $stmt->execute($params);
    if ($stmt->rowCount() > 0) {
        $row = $stmt->fetch(PDO::FETCH_ASSOC);
        if(strcmp($row['password'], hash("sha256",$password)) == 0){
            return true;
        }
    }
    return false;
}
